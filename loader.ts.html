// Unified plugin loader supporting both OpenCode and OpenClaw plugin formats
import { readFile, mkdir, readdir, stat } from 'node:fs/promises';
import path from 'node:path';
import { fileURLToPath } from 'node:url';
import { createHash } from 'node:crypto';
import type { PluginManifest, LoadedPlugin } from '../types/unified.js';
 
export interface PluginLoadOptions {
  pluginPath: string;
  sandbox?: boolean;
  validateOnly?: boolean;
}
 
export interface PluginValidationResult {
  valid: boolean;
  errors: string[];
  warnings: string[];
  manifest: PluginManifest | null;
}
 
// Plugin format detectors
const PluginFormat = {
  OPENTEGRATE: 'opentegrate',
  OPENCODE: 'opencode',
  OPENCLAW: 'openclaw',
} as const;
 
type PluginFormat = typeof PluginFormat[keyof typeof PluginFormat];
 
class PluginLoader {
  private pluginDirs: string[] = [];
  private loadedPlugins: Map<string, LoadedPlugin> = new Map();
  private formatDetectors: Map<PluginFormat, FormatDetector> = new Map();
  private adapters: Map<PluginFormat, PluginAdapter> = new Map();
  
  constructor() {
    this.registerDefaultDetectors();
    this.registerDefaultAdapters();
  }
  
  private registerDefaultDetectors(): void {
    // Opentegrate native format
    this.formatDetectors.set(PluginFormat.OPENTEGRATE, {
      detect: async (dir) => {
        try {
          const manifestPath = path.join(dir, 'plugin.json');
          await stat(manifestPath);
          const content = await readFile(manifestPath, 'utf-8');
          const manifest = JSON.parse(content);
          return manifest.schema === 'opentegrate/v1';
        } catch {
          return false;
        }
      },
      priority: 100,
    });
    
    // OpenCode format (package.json with @opencode/plugin)
    this.formatDetectors.set(PluginFormat.OPENCODE, {
      detect: async (dir) => {
        try {
          const pkgPath = path.join(dir, 'package.json');
          const content = await readFile(pkgPath, 'utf-8');
          const pkg = JSON.parse(content);
          return pkg.name?.startsWith('@opencode/') || pkg.name?.endsWith('-opencode');
        } catch {
          return false;
        }
      },
      priority: 50,
    });
    
    // OpenClaw format (openclaw.plugin.json)
    this.formatDetectors.set(PluginFormat.OPENCLAW, {
      detect: async (dir) => {
        try {
          const manifestPath = path.join(dir, 'openclaw.plugin.json');
          await stat(manifestPath);
          return true;
        } catch {
          return false;
        }
      },
      priority: 50,
    });
  }
  
  private registerDefaultAdapters(): void {
    // Opentegrate native adapter
    this.adapters.set(PluginFormat.OPENTEGRATE, {
      load: async (options) => {
        const manifest = await this.loadManifest(options.pluginPath, PluginFormat.OPENTEGRATE);
        const entry = manifest.main;
        const module = await import(path.join(options.pluginPath, entry));
        
        return {
          manifest,
          instance: module.default ?? module,
          format: PluginFormat.OPENTEGRATE,
        };
      },
      validate: async (manifest) => {
        // Validate against schema
        return { valid: true, errors: [], warnings: [] };
      },
    });
    
    // OpenCode adapter
    this.adapters.set(PluginFormat.OPENCODE, {
      load: async (options) => {
        // Load OpenCode plugin through its package.json
        const pkg = JSON.parse(await readFile(path.join(options.pluginPath, 'package.json'), 'utf-8'));
        const entry = pkg.main ?? pkg.exports?.['.']?.node ?? 'src/index.js';
        
        // Wrap in adapter that implements opentegrate interfaces
        const module = await import(path.join(options.pluginPath, entry));
        
        // Transform OpenCode-specific exports to unified format
        const adapter = this.createOpenCodeAdapter(module, pkg);
        
        return {
          manifest: this.convertOpenCodeManifest(pkg),
          instance: adapter,
          format: PluginFormat.OPENCODE,
        };
      },
      validate: async (manifest) => {
        // OpenCode-specific validation
        return { valid: true, errors: [], warnings: [] };
      },
    });
    
    // OpenClaw adapter
    this.adapters.set(PluginFormat.OPENCLAW, {
      load: async (options) => {
        const manifest = await this.loadManifest(options.pluginPath, PluginFormat.OPENCLAW);
        const entry = manifest.main ?? 'index.js';
        
        // For OpenClaw plugins, we need to import through special loader
        const module = await this.loadOpenClawPlugin(options.pluginPath, entry);
        
        return {
          manifest,
          instance: module,
          format: PluginFormat.OPENCLAW,
        };
      },
      validate: async (manifest) => {
        // OpenClaw-specific validation
        return { valid: true, errors: [], warnings: [] };
      },
    });
  }
  
  async discoverPlugins(directories: string[]): Promise<string[]> {
    const discovered: string[] = [];
    
    for (const dir of directories) {
      try {
        const entries = await readdir(dir, { withFileTypes: true });
        
        for (const entry of entries) {
          if (!entry.isDirectory()) continue;
          
          const pluginPath = path.join(dir, entry.name);
          
          // Try to detect format
          const format = await this.detectFormat(pluginPath);
          if (format) {
            discovered.push(pluginPath);
          }
        }
      } catch (error) {
        console.warn(`Failed to scan directory ${dir}:`, error);
      }
    }
    
    return discovered;
  }
  
  private async detectFormat(pluginPath: string): Promise<PluginFormat | null> {
    const detectors = Array.from(this.formatDetectors.entries())
      .sort((a, b) => b[1].priority - a[1].priority);
    
    for (const [format, detector] of detectors) {
      if (await detector.detect(pluginPath)) {
        return format;
      }
    }
    
    return null;
  }
  
  async loadPlugin(options: PluginLoadOptions): Promise<LoadedPlugin | null> {
    const format = await this.detectFormat(options.pluginPath);
    if (!format) {
      return null;
    }
    
    const adapter = this.adapters.get(format);
    if (!adapter) {
      throw new Error(`No adapter registered for format: ${format}`);
    }
    
    // Validate
    const validation = await adapter.validate(await this.loadManifest(options.pluginPath, format));
    if (!validation.valid) {
      throw new Error(`Plugin validation failed: ${validation.errors.join(', ')}`);
    }
    
    // Load (skip if validateOnly)
    if (options.validateOnly) {
      return {
        manifest: validation.manifest!,
        instance: null,
        format,
      };
    }
    
    const loaded = await adapter.load(options);
    
    // Generate plugin ID from path if not present
    if (!loaded.manifest.id) {
      loaded.manifest.id = this.generatePluginId(loaded.manifest, options.pluginPath);
    }
    
    // Cache loaded plugin
    this.loadedPlugins.set(loaded.manifest.id, loaded);
    
    return loaded;
  }
  
  async loadAllPlugins(pluginDirs: string[], options: Partial<PluginLoadOptions> = {}): Promise<LoadedPlugin[]> {
    const pluginPaths = await this.discoverPlugins(pluginDirs);
    const results: LoadedPlugin[] = [];
    
    for (const pluginPath of pluginPaths) {
      try {
        const plugin = await this.loadPlugin({ ...options, pluginPath });
        if (plugin) {
          results.push(plugin);
        }
      } catch (error) {
        console.error(`Failed to load plugin at ${pluginPath}:`, error);
      }
    }
    
    return results;
  }
  
  getPlugin(id: string): LoadedPlugin | undefined {
    return this.loadedPlugins.get(id);
  }
  
  getAllPlugins(): LoadedPlugin[] {
    return Array.from(this.loadedPlugins.values());
  }
  
  unloadPlugin(id: string): boolean {
    const plugin = this.loadedPlugins.get(id);
    if (plugin?.instance && typeof plugin.instance.dispose === 'function') {
      plugin.instance.dispose();
    }
    return this.loadedPlugins.delete(id);
  }
  
  private async loadManifest(pluginPath: string, format: PluginFormat): Promise<PluginManifest> {
    let manifestPath: string;
    
    switch (format) {
      case PluginFormat.OPENTEGRATE:
        manifestPath = path.join(pluginPath, 'plugin.json');
        break;
      case PluginFormat.OPENCODE:
        manifestPath = path.join(pluginPath, 'package.json');
        break;
      case PluginFormat.OPENCLAW:
        manifestPath = path.join(pluginPath, 'openclaw.plugin.json');
        break;
      default:
        throw new Error(`Unknown format: ${format}`);
    }
    
    const content = await readFile(manifestPath, 'utf-8');
    const manifest = JSON.parse(content);
    
    // Convert to unified format if needed
    if (format !== PluginFormat.OPENTEGRATE) {
      return this.convertToUnifiedManifest(manifest, format);
    }
    
    return manifest;
  }

      permissions: [],

  }
  
  private convertOpenCodeManifest(pkg: any): PluginManifest {
    return {
      id: `opencode-${pkg.name}`,
      name: pkg.displayName ?? pkg.name,
      version: pkg.version,
      description: pkg.description ?? '',
      targets: ['opencode'],
      main: pkg.main ?? 'dist/index.js',
      capabilities: {
        commands: [], // Parse from exports
        extensions: [],
      },
      dependencies: Object.keys(pkg.dependencies ?? {}),
      permissions: [], // Parse from opencode plugin manifest
    };
  }
  
  private async loadOpenClawPlugin(pluginPath: string, entry: string): Promise<unknown> {
    // OpenClaw plugins need special loading with jiti or similar
    // This is a simplified version
    const entryPath = path.join(pluginPath, entry);
    
    // Use dynamic import
    const module = await import(entryPath);
    
    // OpenClaw plugins export specific functions like register
    // We need to wrap them
    return {
      ...module,
      // Add adapter methods
      dispose: () => {
        if (typeof module.unregister === 'function') {
          module.unregister();
        }
      },
    };
  }
  
  private createOpenCodeAdapter(module: any, pkg: any): any {
    // Transform OpenCode module to unified interface
    return {
      ...module,
      dispose: () => {
        if (typeof module.dispose === 'function') {
          module.dispose();
        }
      },
    };
  }
  
  private generatePluginId(manifest: PluginManifest, pluginPath: string): string {
    const base = manifest.name.toLowerCase().replace(/[^a-z0-9]/g, '-');
    const hash = createHash('sha1').update(pluginPath).digest('hex').slice(0, 8);
    return `${base}-${hash}`;
  }
}
 
interface FormatDetector {
  detect(dir: string): Promise<boolean>;
  priority: number;
}
 
interface PluginAdapter {
  load(options: PluginLoadOptions): Promise<{ manifest: PluginManifest; instance: any; format: PluginFormat }>;
  validate(manifest: PluginManifest): Promise<PluginValidationResult>;
}
 
export const pluginLoader = new PluginLoader();
 
// Singleton pattern
export function getPluginLoader(): PluginLoader {
  return pluginLoader;
