import type { ChannelConfig } from '@opentegrate/core';
import type {
  ChannelAttachment,
  ChannelChat,
  ChannelMessage,
  ChannelUser,
  MarkdownFormat,
  SendMessageOptions,
} from './types.js';
import { BaseChannel } from './base-channel.js';
import { createServer, Server as HttpServer, IncomingMessage, ServerResponse } from 'node:http';
import { readFileSync } from 'node:fs';
import path from 'node:path';
 
interface WebChatSession {
  id: string;
  userId: string;
  userName: string;
  connected: boolean;
  ws: WebSocket | null;
  lastActivity: Date;
  messageHistory: ChannelMessage[];
}
 
interface WebChatWidgetConfig {
  title: string;
  subtitle?: string;
  primaryColor: string;
  welcomeMessage: string;
  placeholderText: string;
  showAvatar: boolean;
  avatarUrl?: string;
  position: 'left' | 'right';
}
 
const DEFAULT_WIDGET_CONFIG: WebChatWidgetConfig = {
  title: 'Opentegrate Assistant',
  subtitle: 'AI-powered support',
  primaryColor: '#4F46E5',
  welcomeMessage: 'Hello! How can I help you today?',
  placeholderText: 'Type a message...',
  showAvatar: true,
  position: 'right',
};
 
export class WebChatChannel extends BaseChannel {
  private httpServer: HttpServer | null = null;
  private wss: ReturnType<typeof Bun.serve> | null = null;
  private sessions = new Map<string, WebChatSession>();
  private widgetConfig: WebChatWidgetConfig;
  private staticDir: string;
  private port: number;
  private messageHandler: ((message: ChannelMessage) => Promise<string>) | null = null;
 
  constructor(config: ChannelConfig) {
    super(config, { maxReconnectAttempts: 5, healthCheckInterval: 60000 });
    this.widgetConfig = {
      ...DEFAULT_WIDGET_CONFIG,
      ...(config.connection as unknown as Partial<WebChatWidgetConfig>),
    };
    this.staticDir = config.connection.credentialsPath ?? '/tmp/webchat-static';
    this.port = parseInt(config.connection.apiEndpoint ?? '3002', 10);
  }
 
  async connect(): Promise<void> {
    if (this.connected) return;
 
    try {
      this.startServer();
      this.connected = true;
      this.startTime = Date.now();
      this.health.status = 'healthy';
      this.startHealthChecks();
      this.emit('connected');
    } catch (error) {
      this.recordError(error instanceof Error ? error : new Error(String(error)));
      throw error;
    }
  }
 
  async disconnect(): Promise<void> {
    if (!this.connected) return;
    this.stopServer();
    await this.cleanup();
    this.emit('disconnected');
  }
 
  async sendMessage(chatId: string, content: string, options?: SendMessageOptions): Promise<ChannelMessage> {
    const session = this.sessions.get(chatId);
    if (!session) {
      throw new Error(`Session not found: ${chatId}`);
    }
 
    const message: ChannelMessage = {
      id: `webchat_${this.generateId()}`,
      channelId: this.channelId,
      channelType: 'webchat',
      from: { id: this.channelId, isBot: true, displayName: this.widgetConfig.title },
      chat: { id: chatId, type: 'private', title: session.userName },
      content,
      contentType: 'text',
      timestamp: new Date(),
      replyTo: options?.replyTo,
      attachments: options?.attachments,
      metadata: options?.metadata,
    };
 
    session.messageHistory.push(message);
    this.recordMessageSent();
 
    if (session.ws && session.ws.readyState === 1) {
      session.ws.send(JSON.stringify({
        type: 'message',
        payload: message,
      }));
    }
 
    return message;
  }
 
  async sendTyping(chatId: string): Promise<void> {
    const session = this.sessions.get(chatId);
    if (session?.ws && session.ws.readyState === 1) {
      session.ws.send(JSON.stringify({ type: 'typing' }));
    }
  }
 
  async getUserInfo(userId: string): Promise<ChannelUser> {
    const session = this.sessions.get(userId);
    return {
      id: userId,
      username: session?.userId,
      displayName: session?.userName,
      metadata: { lastActivity: session?.lastActivity },
    };
  }
 
  async getChatHistory(chatId: string, options?: { limit?: number }): Promise<ChannelMessage[]> {
    const session = this.sessions.get(chatId);
    if (!session) return [];
 
    const limit = options?.limit ?? 50;
    return session.messageHistory.slice(-limit);
  }
 
  setOnMessage(handler: (message: ChannelMessage) => Promise<string>): void {
    this.messageHandler = handler;
  }
 
  getActiveSessions(): Array<{ id: string; userId: string; userName: string; connected: boolean; lastActivity: Date }> {
    return Array.from(this.sessions.values()).map(s => ({
      id: s.id,
      userId: s.userId,
      userName: s.userName,
      connected: s.connected,
      lastActivity: s.lastActivity,
    }));
  }
 
  disconnectSession(sessionId: string): void {
    const session = this.sessions.get(sessionId);
    if (session) {
      session.connected = false;
      session.ws?.close();
    }
  }
 
  protected async performHealthCheck(): Promise<void> {
    if (!this.httpServer) {
      throw new Error('HTTP server not running');
    }
  }
 
  protected createFormat(): MarkdownFormat {
    return {
      bold: (text: string) => `**${text}**`,
      italic: (text: string) => `*${text}*`,
      code: (text: string) => `\`${text}\``,
      codeBlock: (text: string, language?: string) => `\`\`\`${language ?? ''}\n${text}\n\`\`\``,
      link: (text: string, url: string) => `[${text}](${url})`,
      list: (items: string[], ordered = false) => items.map((item, i) => ordered ? `${i + 1}. ${item}` : `- ${item}`).join('\n'),
      heading: (text: string, level: number) => `${'#'.repeat(level)} ${text}`,
      blockquote: (text: string) => text.split('\n').map(line => `> ${line}`).join('\n'),
      strikethrough: (text: string) => `~~${text}~~`,
      underline: (text: string) => `__${text}__`,
    };
  }
 
  private startServer(): void {
    this.wss = Bun.serve({
      port: this.port,
      fetch: (req, server) => {
        const url = new URL(req.url);
 
        // WebSocket upgrade
        if (url.pathname === '/ws') {
          const sessionId = url.searchParams.get('session') ?? this.generateId();
          const userId = url.searchParams.get('user') ?? sessionId;
          const userName = url.searchParams.get('name') ?? 'Guest';
 
          const upgraded = server.upgrade(req, { data: { sessionId, userId, userName } });
          if (upgraded) return undefined;
          return new Response('WebSocket upgrade failed', { status: 400 });
        }
 
        // Widget JS
        if (url.pathname === '/widget.js') {
          return new Response(this.generateWidgetScript(), {
            headers: { 'Content-Type': 'application/javascript' },
          });
        }
 
        // Widget HTML
        if (url.pathname === '/widget.html') {
          return new Response(this.generateWidgetHtml(), {
            headers: { 'Content-Type': 'text/html' },
          });
        }
 
        // Health check
        if (url.pathname === '/health') {
          return new Response(JSON.stringify({ status: 'ok', sessions: this.sessions.size }), {
            headers: { 'Content-Type': 'application/json' },
          });
        }
 
        return new Response('Not Found', { status: 404 });
      },
      websocket: {
        open: (ws) => {
          const data = ws.data as { sessionId: string; userId: string; userName: string };
          this.sessions.set(data.sessionId, {
            id: data.sessionId,
            userId: data.userId,
            userName: data.userName,
            connected: true,
            ws: ws as unknown as WebSocket,
            lastActivity: new Date(),
            messageHistory: [],
          });
 
          ws.send(JSON.stringify({
            type: 'connected',
            sessionId: data.sessionId,
            welcome: this.widgetConfig.welcomeMessage,
          }));
        },
        message: async (ws, message) => {
          const data = ws.data as { sessionId: string; userId: string; userName: string };
          const session = this.sessions.get(data.sessionId);
          if (!session) return;
 
          session.lastActivity = new Date();
 
          try {
            const parsed = JSON.parse(message as string);
 
            if (parsed.type === 'message') {
              const userMessage: ChannelMessage = {
                id: `webchat_msg_${this.generateId()}`,
                channelId: this.channelId,
                channelType: 'webchat',
                from: { id: data.userId, displayName: data.userName },
                chat: { id: data.sessionId, type: 'private', title: data.userName },
                content: parsed.content,
                contentType: 'text',
                timestamp: new Date(),
                raw: parsed,
              };
 
              session.messageHistory.push(userMessage);
              this.recordMessageReceived();
              this.emit('message', userMessage);
 
              if (this.messageHandler) {
                await ws.send(JSON.stringify({ type: 'typing' }));
                const response = await this.messageHandler(userMessage);
                await this.sendMessage(data.sessionId, response);
              }
            }
          } catch {
            // Ignore parse errors
          }
        },
        close: (ws) => {
          const data = ws.data as { sessionId: string };
          const session = this.sessions.get(data.sessionId);
          if (session) {
            session.connected = false;
            session.ws = null;
          }
        },
      },
    });
  }
 
  private stopServer(): void {
    if (this.wss) {
      this.wss.stop();
      this.wss = null;
    }
    for (const session of this.sessions.values()) {
      session.connected = false;
    }
  }
