import { promises as fs } from 'node:fs';
import path from 'node:path';
 
export interface Session {
  id: string;
  userId: string;
  channelType: string;
  channelId: string;
  messages: SessionMessage[];
  context: Record<string, unknown>;
  createdAt: Date;
  updatedAt: Date;
  lastActivity: Date;
  metadata: Record<string, unknown>;
}
 
export interface SessionMessage {
  id: string;
  role: 'user' | 'assistant' | 'system';
  content: string;
  timestamp: Date;
  metadata?: Record<string, unknown>;
}
 
export interface SessionSummary {
  id: string;
  userId: string;
  channelType: string;
  messageCount: number;
  createdAt: Date;
  lastActivity: Date;
  topics: string[];
}
 
export class SessionManager {
  private sessions = new Map<string, Session>();
  private userSessions = new Map<string, string[]>();
  private persistPath: string;
  private maxMessagesPerSession: number;
  private maxSessionsPerUser: number;
  private sessionTtlMs: number;
 
  constructor(options: {
    persistPath?: string;
    maxMessagesPerSession?: number;
    maxSessionsPerUser?: number;
    sessionTtlMs?: number;
  } = {}) {
    this.persistPath = options.persistPath ?? '/tmp/opentegrate-sessions.json';
    this.maxMessagesPerSession = options.maxMessagesPerSession ?? 200;
    this.maxSessionsPerUser = options.maxSessionsPerUser ?? 50;
    this.sessionTtlMs = options.sessionTtlMs ?? 7 * 24 * 60 * 60 * 1000;
  }
 
  createSession(userId: string, channelType: string, channelId: string, options?: { context?: Record<string, unknown>; metadata?: Record<string, unknown> }): Session {
    const userSessionIds = this.userSessions.get(userId) ?? [];
 
    if (userSessionIds.length >= this.maxSessionsPerUser) {
      const oldestId = userSessionIds[0];
      this.deleteSession(oldestId);
    }
 
    const session: Session = {
      id: `session_${Date.now()}_${Math.random().toString(36).slice(2, 8)}`,
      userId,
      channelType,
      channelId,
      messages: [],
      context: options?.context ?? {},
      createdAt: new Date(),
      updatedAt: new Date(),
      lastActivity: new Date(),
      metadata: options?.metadata ?? {},
    };
 
    this.sessions.set(session.id, session);
    userSessionIds.push(session.id);
    this.userSessions.set(userId, userSessionIds);
 
    return session;
  }
 
  getSession(sessionId: string): Session | null {
    const session = this.sessions.get(sessionId);
    if (!session) return null;
 
    if (Date.now() - session.lastActivity.getTime() > this.sessionTtlMs) {
      this.deleteSession(sessionId);
      return null;
    }
 
    return session;
  }
 
  getUserSessions(userId: string): Session[] {
    const sessionIds = this.userSessions.get(userId) ?? [];
    return sessionIds
      .map(id => this.sessions.get(id))
      .filter((s): s is Session => s !== null && Date.now() - s.lastActivity.getTime() <= this.sessionTtlMs);
  }
 
  addMessage(sessionId: string, message: Omit<SessionMessage, 'id' | 'timestamp'>): SessionMessage | null {
    const session = this.getSession(sessionId);
    if (!session) return null;
 
    const sessionMessage: SessionMessage = {
      ...message,
      id: `msg_${Date.now()}_${Math.random().toString(36).slice(2, 6)}`,
      timestamp: new Date(),
    };
 
    session.messages.push(sessionMessage);
 
    if (session.messages.length > this.maxMessagesPerSession) {
      const systemMessages = session.messages.filter(m => m.role === 'system');
      const recentMessages = session.messages.filter(m => m.role !== 'system').slice(-this.maxMessagesPerSession + systemMessages.length);
      session.messages = [...systemMessages, ...recentMessages];
    }
 
    session.updatedAt = new Date();
    session.lastActivity = new Date();
 
    return sessionMessage;
  }
 
  getContext(sessionId: string, maxMessages?: number): { messages: SessionMessage[]; context: Record<string, unknown> } | null {
    const session = this.getSession(sessionId);
    if (!session) return null;
 
    const messages = maxMessages
      ? session.messages.slice(-maxMessages)
      : session.messages;
 
    return { messages, context: session.context };
  }
 
  setContext(sessionId: string, context: Record<string, unknown>): boolean {
    const session = this.getSession(sessionId);
    if (!session) return false;
    session.context = { ...session.context, ...context };
    session.updatedAt = new Date();
    return true;
  }
 
  deleteSession(sessionId: string): boolean {
    const session = this.sessions.get(sessionId);
    if (!session) return false;
 
    const userSessionIds = this.userSessions.get(session.userId) ?? [];
    const index = userSessionIds.indexOf(sessionId);
    if (index !== -1) {
      userSessionIds.splice(index, 1);
      this.userSessions.set(session.userId, userSessionIds);
    }
 
    return this.sessions.delete(sessionId);
  }
 
  linkSessions(sessionId1: string, sessionId2: string): boolean {
    const s1 = this.getSession(sessionId1);
    const s2 = this.getSession(sessionId2);
    if (!s1 || !s2) return false;
 
    s1.metadata.linkedSession = sessionId2;
    s2.metadata.linkedSession = sessionId1;
 
    return true;
  }
 
  async searchSessions(query: string, options?: { userId?: string; channelType?: string; limit?: number }): Promise<SessionSummary[]> {
    const lower = query.toLowerCase();
    const limit = options?.limit ?? 20;
    const results: SessionSummary[] = [];
 
    for (const session of this.sessions.values()) {
      if (options?.userId && session.userId !== options.userId) continue;
      if (options?.channelType && session.channelType !== options.channelType) continue;
 
      const hasMatch = session.messages.some(m =>
        m.content.toLowerCase().includes(lower)
      );
 
      if (hasMatch || lower === '') {
        const topics = this.extractTopics(session);
        results.push({
          id: session.id,
          userId: session.userId,
          channelType: session.channelType,
          messageCount: session.messages.length,
          createdAt: session.createdAt,
          lastActivity: session.lastActivity,
          topics,
        });
      }
 
      if (results.length >= limit) break;
    }
 
    return results.sort((a, b) => b.lastActivity.getTime() - a.lastActivity.getTime());
  }
 
  exportSession(sessionId: string, format: 'json' | 'markdown' | 'text'): string | null {
    const session = this.getSession(sessionId);
    if (!session) return null;
 
    switch (format) {
      case 'json':
        return JSON.stringify(session, null, 2);
 
      case 'markdown': {
        let md = `# Session: ${session.id}\n\n`;
        md += `**User:** ${session.userId}\n`;
        md += `**Channel:** ${session.channelType}\n`;
        md += `**Created:** ${session.createdAt.toISOString()}\n`;
        md += `**Messages:** ${session.messages.length}\n\n`;
        md += `---\n\n`;
 
        for (const msg of session.messages) {
          const role = msg.role === 'user' ? '👤 User' : '🤖 Assistant';
          md += `### ${role} (${msg.timestamp.toISOString()})\n\n`;
          md += `${msg.content}\n\n`;
        }
 
        return md;
      }
 
      case 'text': {
        let text = `Session: ${session.id}\n`;
        text += `User: ${session.userId}\n`;
        text += `Channel: ${session.channelType}\n`;
        text += `${'='.repeat(50)}\n\n`;
 
        for (const msg of session.messages) {
          const role = msg.role === 'user' ? 'User' : 'Assistant';
          text += `[${msg.timestamp.toISOString()}] ${role}:\n${msg.content}\n\n`;
        }
 
        return text;
      }
 
      default:
        return null;
    }
  }
 
  async persist(): Promise<void> {
    const data = {
      sessions: Array.from(this.sessions.values()),
      userSessions: Object.fromEntries(this.userSessions),
    };
 
    await fs.mkdir(path.dirname(this.persistPath), { recursive: true });
    await fs.writeFile(this.persistPath, JSON.stringify(data, null, 2));
  }
 
  async restore(): Promise<void> {
    try {
      const content = await fs.readFile(this.persistPath, 'utf-8');
      const data = JSON.parse(content);
 
      for (const sessionData of data.sessions ?? []) {
        const session: Session = {
          ...sessionData,
          createdAt: new Date(sessionData.createdAt),
          updatedAt: new Date(sessionData.updatedAt),
          lastActivity: new Date(sessionData.lastActivity),
          messages: sessionData.messages.map((m: any) => ({
            ...m,
            timestamp: new Date(m.timestamp),
          })),
        };
        this.sessions.set(session.id, session);
      }
 
      this.userSessions = new Map(Object.entries(data.userSessions ?? {}));
    } catch {
      // No persisted sessions to restore
    }
  }
 
  cleanup(): void {
    const now = Date.now();
    const expiredSessions: string[] = [];
 
    for (const [id, session] of this.sessions) {
      if (now - session.lastActivity.getTime() > this.sessionTtlMs) {
        expiredSessions.push(id);
      }
    }
 
    for (const id of expiredSessions) {
      this.deleteSession(id);
    }
  }
 
  getStats(): { total: number; active: number; totalMessages: number; uniqueUsers: number } {
    const now = Date.now();
    const active = Array.from(this.sessions.values()).filter(
      s => now - s.lastActivity.getTime() < 3600000
    ).length;
 
    const totalMessages = Array.from(this.sessions.values()).reduce(
      (sum, s) => sum + s.messages.length, 0
    );
 
    return {
      total: this.sessions.size,
      active,
      totalMessages,
      uniqueUsers: this.userSessions.size,
    };
  }
 
  private extractTopics(session: Session): string[] {
    const topics = new Set<string>();
    const content = session.messages.map(m => m.content).join(' ').toLowerCase();
 
    const topicPatterns = [
      /code|programming|function|class|api/i,
      /weather|temperature|forecast/i,
      /help|support|question/i,
      /greeting|hello|hi|hey/i,
      /search|find|look/i,
      /email|message|communication/i,
      /file|document|storage/i,
      /config|setting|setup/i,
    ];
 
    const topicNames = ['coding', 'weather', 'help', 'greeting', 'search', 'email', 'files', 'config'];
 
    for (let i = 0; i < topicPatterns.length; i++) {
      if (topicPatterns[i].test(content)) {
        topics.add(topicNames[i]);
      }
    }
 
    return Array.from(topics);
  }
}
 
let sessionManagerInstance: SessionManager | null = null;
 
export function getSessionManager(): SessionManager {
  if (!sessionManagerInstance) {
    sessionManagerInstance = new SessionManager();
  }
  return sessionManagerInstance;
}
 
export function resetSessionManager(): void {
  sessionManagerInstance = null;
}
